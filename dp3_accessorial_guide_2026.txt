    <div class="section-header" id="item105">
      <div class="ico">&#128230;</div>
      <h2>Packing, Unpacking, and Crating Services</h2>
      <div class="item-code">400NG Item 105</div>
    </div>
    <div class="section-body">
      <div class="reg-refs">
        <span class="reg-badge reg-400ng">400NG Item 105A/105B/105D/105J</span>
        <span class="reg-badge reg-dtr">DTR Part IV Ch. 402 App. B</span>
        <span class="reg-badge reg-jtr">JTR Par. 051002</span>
      </div>

      <div class="callout callout-gold">
        <h4>&#9989; Pre-Approval Required</h4>
        <p>Pack, Crate, Unpack, and Debris Removal all require <strong>PPSO pre-approval</strong> before service is performed, or must be Government-requested. Verify DPS pre-approval entry before paying any invoice.</p>
      </div>

      <div class="tab-row">
        <button class="tab active" onclick="showTab(this,'tabs-105','t105-overview')">Overview</button>
        <button class="tab" onclick="showTab(this,'tabs-105','t105-formula')">Formulas</button>
        <button class="tab" onclick="showTab(this,'tabs-105','t105-example')">Examples</button>
        <button class="tab" onclick="showTab(this,'tabs-105','t105-validate')">Validation</button>
        <button class="tab" onclick="showTab(this,'tabs-105','t105-crating')">Crating Rules</button>
      </div>
      <div id="tabs-105">

      <div id="t105-overview" class="tab-panel active">
        <table>
          <tr><th>Item Code</th><th>Service</th><th>Basis</th><th>Discount</th></tr>
          <tr><td>105A</td><td>Full Pack / Full Unpack</td><td>Net weight (CWT)</td><td>dLHS applies</td></tr>
          <tr><td>105B</td><td>Pack / Unpack Regular Crate</td><td>Gross cu ft. of crate (min 4 cu ft.)</td><td>dLHS applies</td></tr>
          <tr><td>105D</td><td>Debris Removal (within 30 days of delivery)</td><td>Net weight (CWT)</td><td>dLHS applies</td></tr>
          <tr><td>105J</td><td>Storage Inspection Fee (in lieu of 105A Full Pack)</td><td>Actual net CWT</td><td>No dLHS on 105J; dLHS applies to unpack</td></tr>
        </table>
        <p><strong>Packing/Crating basis:</strong> Rate applied at <em>origin</em> Service Area (block 19 of BL). <strong>Unpacking/Uncrating basis:</strong> Rate applied at <em>destination</em> Service Area (block 18 of BL).</p>
        <p><strong>Full Unpack Rate:</strong> Always calculated at the rate line for Full Pack CWT ≤ 16,000 lbs at the <em>destination</em> Service Schedule × 25% × InvdLHS (58% fixed discount per tariff).</p>
        <div class="callout callout-info">
          <h4>Vehicle Weight Deduction</h4>
          <p>Motor passenger vehicle weight (auto, pickup, van, SUV, dune buggy) must be <strong>deducted from shipment weight</strong> before computing 105A Full Pack charges. Use NADA or manufacturer's weight.</p>
        </div>
        <div class="callout callout-warn">
          <h4>105J vs 105A — Mutually Exclusive</h4>
          <p>Do NOT bill both 105A and 105J on the same shipment. Use 105J only when shipment picks up from a storage warehouse and is already packed. The Full Unpack (billed as Misc item code 226A) is still owed and subject to dLHS.</p>
        </div>
      </div>

      <div id="t105-formula" class="tab-panel">
        <h3>Full Pack Formula</h3>
        <div class="formula">105A Pack = (Pack Rate × CWT) × InvdLHS</div>
        <h3>Full Unpack Formula</h3>
        <div class="formula">105A Unpack = (Unpack Rate × CWT) × InvdLHS

  Subject to 4 cu ft. minimum; truncate at 2 decimal places (no rounding up)</div>
        <h3>Debris Removal (105D)</h3>

  Only applies when performed AFTER delivery date and within 30 days</div>
      </div>

    <div class="section-header" id="item16">
      <div class="ico">&#9981;</div>
      <h2>Fuel Surcharge (FSC) / Fuel Related Rate Adjustment (FRA)</h2>
      <div class="item-code">400NG Item 16</div>
    </div>
    <div class="section-body">
      <div class="reg-refs">
        <span class="reg-badge reg-400ng">400NG Item 16A / 16B</span>
        <span class="reg-badge reg-dtr">DTR Part IV Ch. 402</span>
      </div>
      <table>
        <tr><th>Code</th><th>Service</th></tr>
        <tr><td>16A</td><td>Fuel Surcharge — Linehaul (LHS)</td></tr>
        <tr><td>16B</td><td>Fuel Surcharge — Delivery to/from SIT</td></tr>
      </table>

      <h3>FRA Calculation</h3>
      <p><strong>Baseline fuel price:</strong> $3.50/gallon. For every increment of $0.13 above $3.50, TSP is entitled to <strong>1% of the linehaul rate</strong>.</p>
      <p>FRA is determined by the national average diesel fuel price published on the <strong>first Monday of each month</strong> (or first business day after Monday if Monday is a holiday) by DOE/EIA at <a href="https://www.eia.gov/petroleum/gasdiesel/" target="_blank">EIA website</a>.</p>

      <div class="example-block">
        <h4>&#128221; FRA Example: EIA price = $5.15/gallon</h4>
        <div class="calc-line">Overage = $5.15 − $3.50 = $1.65</div>
        <div class="calc-line">Increments = $1.65 ÷ $0.13 = 12.69 → 12 complete increments = <strong>12% FRA</strong></div>
        <div class="result">TSP entitled to 12% of the linehaul rate as fuel surcharge</div>
      </div>

      <div class="callout callout-warn">
        <h4>FSC NOT Authorized On:</h4>
        <ul>
          <li>Spot bid or One-Time-Only (OTO) personal property movements (any mode)</li>
          <li>Rail, barge, or pipeline movements</li>
          <li>Third party service charges</li>
          <li>Alaska shipments picked up/delivered within 14 miles of nearest BPC water port</li>
        </ul>
      </div>

      <p>TSP must apply the linehaul or SIT discount to baseline charges before billing item codes 16A or 16B. The current FSC table is posted at <a href="https://www.ustranscom.mil/dp3/pdfs.cfm" target="_blank">USTRANSCOM DP3 website</a>.</p>

      <h3>&#128270; Technician Validation Checklist — FSC/FRA</h3>
      <ul class="checklist">
        <li>Confirm EIA diesel fuel price from the first Monday of the month in which shipment was picked up</li>
        <li>Confirm FRA percentage from USTRANSCOM FSC table</li>
        <li>Confirm shipment is not an OTO, rail, barge, pipeline, or third-party movement</li>
        <li>Confirm linehaul or SIT discount was applied to baseline before FSC computed</li>
        <li>For Alaska: confirm pickup/delivery is MORE than 14 miles from BPC water port before approving FSC</li>
      </ul>
    </div>

      <p>Waiting time cannot apply when it is the fault of the TSP. Charges apply beyond free waiting time allowances.</p>
      <table>
        <tr><th>Item</th><th>Service</th></tr>
        <tr><td>503A</td><td>Vehicle waiting time ($32.29/hr per vehicle)</td></tr>
        <tr><td>503B</td><td>Regular labor waiting time (uses Item 502 rates)</td></tr>
        <tr><td>503C</td><td>OT labor waiting time (uses Item 502 OT rates)</td></tr>
        <tr><td>503D</td><td>Sunday/holiday labor (Alaska only)</td></tr>
      </table>
      <h3>Free Waiting Time</h3>
      <ul>
        <li>1 hour at origin (pickup)</li>
        <li>1 hour for direct deliveries to residence (PPSO notified waiting time applies after)</li>
      </ul>
      <h3>&#128270; Technician Validation Checklist — IT-26 Waiting Time</h3>
      <ul class="checklist">
        <li>Confirm waiting time is not the fault of the TSP</li>
        <li>Confirm free waiting time has expired before charging (1 hr at origin; 1 hr direct delivery)</li>
        <li>Confirm 503A vehicle charge is supported by driver log or vehicle record</li>
        <li>Confirm labor charges (503B/C) use IT-26 Item 502 rates for applicable location</li>
        <li>Alaska Sunday/holiday (503D): confirm service was actually performed on Sunday or recognized holiday in Alaska</li>
      </ul>
    </div>

function showTab(btn, containerID, panelID) {
  const container = document.getElementById(containerID);
  container.querySelectorAll('.tab-panel').forEach(p => p.classList.remove('active'));

// ── SEARCH ──
function doSearch(query) {
  const msg = document.getElementById('search-results-msg');
  // Remove previous highlights
  document.querySelectorAll('mark').forEach(m => {
    m.outerHTML = m.innerHTML;
  });
  if (!query || query.trim().length < 2) {
    msg.style.display = 'none';
    return;
  }
  const q = query.trim();
  const regex = new RegExp('(' + q.replace(/[.*+?^${}()|[\]\\]/g, '\\$&') + ')', 'gi');
  let count = 0;
  const walker = document.createTreeWalker(document.getElementById('main'), NodeFilter.SHOW_TEXT, null, false);
  const nodes = [];
  while (walker.nextNode()) nodes.push(walker.currentNode);
  nodes.forEach(node => {
    if (node.parentNode.nodeName === 'SCRIPT' || node.parentNode.nodeName === 'STYLE') return;
    if (regex.test(node.textContent)) {
      const span = document.createElement('span');
      span.innerHTML = node.textContent.replace(regex, '<mark>$1</mark>');
      count += (node.textContent.match(regex) || []).length;
      node.parentNode.replaceChild(span, node);
    }
  });
  if (count > 0) {
    msg.textContent = '🔍 Found ' + count + ' match' + (count>1?'es':'') + ' for "' + q + '". Scroll to review highlighted results.';
    msg.style.display = 'block';
    // Scroll to first match
    const first = document.querySelector('mark');
    if (first) first.scrollIntoView({ behavior: 'smooth', block: 'center' });
  } else {
    msg.textContent = '⚠ No matches found for "' + q + '".';
    msg.style.display = 'block';
  }
}
